# App::Parrot::Create

# Depends on
    Mojolicious
    Mojolicious::Plugin::YamlConfig
    Mojolicious::Plugin::RenderFile
    

# Installing Dependencies

With cpanminus

    cpanm --installdeps .

# Running tests

    make test

# Cleaning up

    make clean

# INSTALL

    perl Makefile.PL
    make install
    
# RUN #
    perl app-parrot-create daemon
    
    or
    
    morbo app-parrot-create
    
# VIEW #
    http://127.0.0.1:3000
